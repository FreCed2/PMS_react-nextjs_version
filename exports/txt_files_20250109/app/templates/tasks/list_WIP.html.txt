<script>
    /**
     * Highlights a task row with a flash animation.
     * @param {string} taskId - The ID of the task to highlight.
     */
    function highlightTask(taskId) {
        console.log("Highlighting task with ID:", taskId); // Debugging

        // Select the task row
        const taskRow = document.getElementById(`task-${taskId}`);
        console.log("Task row element:", taskRow);

        if (taskRow) {
            console.log("Task row found:", taskRow); // Debugging

            // Add the highlight class with a slight delay
            setTimeout(() => {
                taskRow.classList.add("highlight");

                // Remove the highlight class after the animation
                setTimeout(() => {
                    taskRow.classList.remove("highlight");
                    console.log("Highlight removed for task ID:", taskId);
                }, 1000); // Match this duration to the CSS animation time
            }, 500);
        } else {
            console.error("Task row not found for ID:", taskId); // Debugging
        }
    }
</script>
