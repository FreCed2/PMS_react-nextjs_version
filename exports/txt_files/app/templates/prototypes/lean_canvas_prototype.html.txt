  <script>
    // Stub function for AI suggestions.
    // In a real integration, this would call your AI service API.
    function getAISuggestion(fieldId) {
      let suggestion = "";
      switch (fieldId) {
        case "businessOutcomes":
          suggestion = "more engaged users and higher conversion rates";
          break;
        case "usersCustomers":
          suggestion = "tech-savvy early adopters";
          break;
        case "userBenefits":
          suggestion = "a seamless and intuitive experience";
          break;
        case "solutionIdeas":
          suggestion = "a streamlined dashboard with real-time analytics";
          break;
        default:
          suggestion = "";
      }
      document.getElementById(fieldId).value = suggestion;
    }

    // Combines responses from questions 2, 3, 4, and 5 into the hypothesis statement.
    function generateHypothesis() {
      const businessOutcomes = document.getElementById('businessOutcomes').value.trim();
      const usersCustomers   = document.getElementById('usersCustomers').value.trim();
      const userBenefits     = document.getElementById('userBenefits').value.trim();
      const solutionIdeas    = document.getElementById('solutionIdeas').value.trim();

      // Create the hypothesis statement using the template:
      // "We believe that [business outcome] will be achieved if [user] attains [benefit] with [feature]."
      const hypothesis = `We believe that ${businessOutcomes} will be achieved if ${usersCustomers} attains ${userBenefits} with ${solutionIdeas}.`;

      document.getElementById('hypothesisResult').innerText = hypothesis;
    }
